Используемые команды:
clear - очистка экрана терминала
reset - перезагрузка
pwd - локация
cd .. - в директорию выше
cd -  - в предыдущую директорию
cd ~  - в домашний каталог
Путь cd через </> - абсолютный, не через </> - относительный
ls - просмотр содержимого директории
ls -l  - содержимое в таблице
ls -a  - +скрытые файлы 
ll, ls -la/al/-l-a/-a-l - максимально подробное содержимое
mkdir - создание директорий
mkdir folder1 folder2 - 2 папки в одной директории
mkdir folder1/folder2 - папка в папке
tree - просмотр дерева каталогов и файлов
touch - новый файл
echo 'текст' >file1.txt - создание файла и запись текста в него
cat >file1.txt - создание файла с последующим наполнением (выход и сохранение ctrl+D)
cp - копирование: cp file folder/
cp -r - рекурсивное (для директорий)
mv - перемещение: mv file folder/ или folder1/ folder2/
   - переименование: mv file1 file11 --> остаётся только file11
rm - удаление файлов
rm -r удаление директорий

редкор nano:
ctrl+X - выход
ctrl+О - запись
nano file -B - перезапись с сохранением файла BackUp

редактор vim:
:q! - выйти без вопросов
I - режим INSERT
ESC - выход из режима редактирования
:w - cохранить
:q - выход
:wq - записать+выйти
